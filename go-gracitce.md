### for loop:
```go
for initialization; condition; update {
  statement(s)
}
```
### if else condition:
```go
if condition {
  // code
} else {
  // code
}
```

switch expression {
  case <value1>:
    // code block 1

  case <value2>:
    // code block 2

  case <value3>:
    // code block 3
  ...

  default:
